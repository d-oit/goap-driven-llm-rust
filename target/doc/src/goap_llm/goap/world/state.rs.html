1//! World state implementation
2//!
3//! The WorldState tracks all relevant information during planning and execution.
4
5use crate::goap::world::property::WorldProperty;
6use serde::{Deserialize, Serialize};
7use std::collections::HashSet;
8use std::time::{Duration, Instant};
9
10/// Represents the current state of the world
11#[derive(Debug, Clone)]
12pub struct WorldState {
13    /// Set of true properties in the world
14    properties: HashSet<WorldProperty>,
15
16    /// Total token budget for this request
17    token_budget: u32,
18
19    /// Tokens used so far
20    tokens_used: u32,
21
22    /// Start time of planning/execution
23    start_time: Instant,
24
25    /// Current step in the execution
26    current_step: u32,
27
28    /// Request being processed
29    request: String,
30}
31
32impl WorldState {
33    /// Create a new world state
34    pub fn new(token_budget: u32, request: String) -> Self {
35        WorldState {
36            properties: HashSet::new(),
37            token_budget,
38            tokens_used: 0,
39            start_time: Instant::now(),
40            current_step: 0,
41            request,
42        }
43    }
44
45    /// Check if a property is true
46    pub fn has_property(&self, property: &WorldProperty) -> bool {
47        self.properties.contains(property)
48    }
49
50    /// Set a property to true
51    pub fn set_property(&mut self, property: WorldProperty) {
52        self.properties.insert(property);
53    }
54
55    /// Remove a property
56    pub fn remove_property(&self, property: &WorldProperty) -> bool {
57        false // TODO: Implement if needed
58    }
59
60    /// Get all current properties
61    pub fn get_properties(&self) -> &HashSet<WorldProperty> {
62        &self.properties
63    }
64
65    /// Check if token budget is available
66    pub fn tokens_available(&self) -> bool {
67        self.tokens_used < self.token_budget
68    }
69
70    /// Get remaining tokens
71    pub fn tokens_remaining(&self) -> u32 {
72        self.token_budget.saturating_sub(self.tokens_used)
73    }
74
75    /// Add tokens used
76    pub fn add_tokens_used(&mut self, tokens: u32) {
77        self.tokens_used = self.tokens_used.saturating_add(tokens);
78    }
79
80    /// Get total token budget
81    pub fn get_token_budget(&self) -> u32 {
82        self.token_budget
83    }
84
85    /// Get tokens used
86    pub fn get_tokens_used(&self) -> u32 {
87        self.tokens_used
88    }
89
90    /// Get elapsed time
91    pub fn elapsed(&self) -> Duration {
92        self.start_time.elapsed()
93    }
94
95    /// Get current step
96    pub fn get_current_step(&self) -> u32 {
97        self.current_step
98    }
99
100    /// Increment step
101    pub fn increment_step(&mut self) {
102        self.current_step += 1;
103    }
104
105    /// Get the request
106    pub fn get_request(&self) -> &str {
107        &self.request
108    }
109
110    /// Check if a state satisfies a goal
111    pub fn satisfies(&self, properties: &[WorldProperty]) -> bool {
112        properties.iter().all(|prop| self.has_property(prop))
113    }
114
115    /// Get the difference between this state and another
116    pub fn difference<'a>(&self, other: &'a WorldState) -> Vec<WorldProperty> {
117        other.properties
118            .difference(&self.properties)
119            .cloned().collect()
120    }
121}
122
123impl Default for WorldState {
124    fn default() -> Self {
125        Self::new(10000, String::new())
126    }
127}
