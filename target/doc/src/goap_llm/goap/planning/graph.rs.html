1//! Action graph implementation
2//!
3//! Represents the graph of available actions and their relationships.
4
5use crate::goap::actions::{Action, ActionType};
6use crate::goap::world::property::WorldProperty;
7use serde::{Deserialize, Serialize};
8use std::collections::{HashMap, HashSet};
9use std::hash::{Hash, Hasher};
10use std::cmp::Eq;
11
12/// Represents the action graph for planning
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct ActionGraph {
15    /// Map of action types to their definitions
16    actions: HashMap<ActionType, Action>,
17
18    /// Adjacency list: which actions can follow which
19    transitions: HashMap<ActionType, HashSet<ActionType>>,
20
21    /// Available actions in the graph
22    available_actions: Vec<ActionType>,
23}
24
25impl ActionGraph {
26    /// Create a new empty action graph
27    pub fn new() -> Self {
28        ActionGraph {
29            actions: HashMap::new(),
30            transitions: HashMap::new(),
31            available_actions: Vec::new(),
32        }
33    }
34
35    /// Add an action to the graph
36    pub fn add_action(&mut self, action: Action) -> &mut Self {
37        let action_type = action.action_type.clone();
38        self.actions.insert(action_type.clone(), action);
39        if !self.available_actions.contains(&action_type) {
40            self.available_actions.push(action_type);
41        }
42        self
43    }
44
45    /// Add a transition between actions
46    pub fn add_transition(&mut self, from: ActionType, to: ActionType) -> &mut Self {
47        self.transitions
48            .entry(from)
49            .or_insert_with(HashSet::new)
50            .insert(to);
51        self
52    }
53
54    /// Get an action by type
55    pub fn get_action(&self, action_type: &ActionType) -> Option<&Action> {
56        self.actions.get(action_type)
57    }
58
59    /// Get all available actions
60    pub fn get_actions(&self) -> &[ActionType] {
61        &self.available_actions
62    }
63
64    /// Get actions that can follow the given action
65    pub fn get_next_actions(&self, action: &ActionType) -> Vec<ActionType> {
66        self.transitions
67            .get(action)
68            .map(|set| set.iter().cloned().collect())
69            .unwrap_or_default()
70    }
71
72    /// Get all actions in the graph
73    pub fn get_all_actions(&self) -> HashMap<ActionType, Action> {
74        self.actions.clone()
75    }
76
77    /// Build a default action graph with standard actions
78    pub fn build_default() -> Self {
79        let mut graph = ActionGraph::new();
80
81        // Add all standard actions with their preconditions and effects
82
83        // 1. Detect schema type
84        graph.add_action(
85            Action::new(ActionType::DetectSchemaType)
86                .with_precondition(WorldProperty::RequestValidated)
87                .with_effect(WorldProperty::SchemaTypeDetected)
88                .with_cost(10)
89                .with_duration(50),
90        );
91
92        // 2. Fetch schema
93        graph.add_action(
94            Action::new(ActionType::FetchSchema)
95                .with_precondition(WorldProperty::SchemaTypeDetected)
96                .with_effect(WorldProperty::SchemaAvailable)
97                .with_cost(20)
98                .with_duration(100),
99        );
100
101        // 3. Check pattern cache
102        graph.add_action(
103            Action::new(ActionType::CheckPatternCache)
104                .with_precondition(WorldProperty::RequestValidated)
105                .with_effect(WorldProperty::PatternCacheChecked)
106                .with_cost(5)
107                .with_duration(20),
108        );
109
110        // 4. Generate response
111        graph.add_action(
112            Action::new(ActionType::GenerateResponse)
113                .with_precondition(WorldProperty::RequestValidated)
114                .with_effect(WorldProperty::ResponseGenerated)
115                .with_cost(100)
116                .with_duration(500),
117        );
118
119        // 5. Generate from pattern (if available)
120        graph.add_action(
121            Action::new(ActionType::GenerateFromPattern)
122                .with_precondition(WorldProperty::PatternAvailable("found".to_string()))
123                .with_effect(WorldProperty::ResponseGenerated)
124                .with_cost(30)
125                .with_duration(150),
126        );
127
128        // 6. Post-validate response
129        graph.add_action(
130            Action::new(ActionType::PostValidateResponse)
131                .with_precondition(WorldProperty::ResponseGenerated)
132                .with_effect(WorldProperty::ResponseValidated)
133                .with_cost(15)
134                .with_duration(75),
135        );
136
137        // 7. Learn success pattern
138        graph.add_action(
139            Action::new(ActionType::LearnSuccessPattern)
140                .with_precondition(WorldProperty::ResponseValidated)
141                .with_effect(WorldProperty::LearnSuccessPattern)
142                .with_cost(10)
143                .with_duration(50),
144        );
145
146        // Define transitions
147        graph.add_transition(ActionType::DetectSchemaType, ActionType::FetchSchema);
148        graph.add_transition(ActionType::FetchSchema, ActionType::GenerateResponse);
149        graph.add_transition(ActionType::CheckPatternCache, ActionType::GenerateFromPattern);
150        graph.add_transition(ActionType::GenerateResponse, ActionType::PostValidateResponse);
151        graph.add_transition(ActionType::GenerateFromPattern, ActionType::PostValidateResponse);
152        graph.add_transition(ActionType::PostValidateResponse, ActionType::LearnSuccessPattern);
153
154        graph
155    }
156}
157
158impl Default for ActionGraph {
159    fn default() -> Self {
160        Self::build_default()
161    }
162}

164#[cfg(test)]
165mod tests {
166    use super::*;
167
168    #[test]
169    fn test_action_graph_creation() {
170        let graph = ActionGraph::new();
171        assert!(graph.get_all_actions().is_empty());
172    }
173
174    #[test]
175    fn test_add_action() {
176        let mut graph = ActionGraph::new();
177        let action = Action::new(ActionType::GenerateResponse);
178        graph.add_action(action);
179
180        assert_eq!(graph.get_actions().len(), 1);
181        assert!(graph.get_action(&ActionType::GenerateResponse).is_some());
182    }
183
184    #[test]
185    fn test_add_transition() {
186        let mut graph = ActionGraph::new();
187        graph.add_transition(ActionType::GenerateResponse, ActionType::PostValidateResponse);
188
189        let next_actions = graph.get_next_actions(&ActionType::GenerateResponse);
190        assert!(next_actions.contains(&ActionType::PostValidateResponse));
191    }
192
193    #[test]
194    fn test_build_default() {
195        let graph = ActionGraph::build_default();
196        assert!(!graph.get_actions().is_empty());
197        assert!(graph.get_actions().contains(&ActionType::GenerateResponse));
198    }
